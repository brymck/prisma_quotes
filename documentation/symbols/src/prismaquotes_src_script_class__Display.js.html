 44 	/**
 45 	 * Adds commas to a number
 46 	 * @param {string} num An unformatted number
 47 	 * @return The formatted number
 48 	 * @type string
 49 	 */
 50 	function addCommas(num){
 51 		var temp = num.toString().split('.');
 52 		while (DECIMAL_REGEX.test(temp[0])) {
 53 			temp[0] = temp[0].replace(DECIMAL_REGEX, '$1,$2');
 54 		}
 55 		return temp.join('.');
 56 	};

 58 	/**
 59 	 * Compares the text content of two jQuery nodes, returning "up" if the next
 60 	 * node has a higher value than the previous node, "down" if it's lower, and
 61 	 * "" if there's no difference or one node contains a value that's not a
 62 	 * number   
 63 	 * @param {Object} $prev The previous jQuery-wrapped node
 64 	 * @param {Object} $next The next jQuery-wrapped node
 65 	 * @return A text result showing how the value has changed from the previous
 66 	 * to the next node
 67 	 * @type string 
 68 	 */
 69 	function compareValues($prev, $next) {
 70 		// Parse as floating-point numbers, noting that parseFloat return NaN for
 71 		// non-number strings
 72 		var prev = parseFloat($prev.text().replace(/,/g, ''));
 73 		var next = parseFloat($next.text().replace(/,/g, ''));
 74 		if (isNaN(prev) || isNaN(next) || prev === next) {
 75 			return '';
 76 		} else {
 77 			return (prev < next ? 'up' : 'down');
 78 		}
 79 	}

137 			/**
138 			 * Creates a formatted cell containing security information
139 			 * @param {Object} quote Full information on the quote
140 			 * @param {Object} metric The metric used to evaluate the quote
141 			 * @return Node
142 			 * @type object
143 			 */
144 			cell: function(quote, metric){
145 				// Creates 
146 				var $td = $('<td>').addClass(metric.type);
147 				try {
148 					var value = metric.func.call(quote);
149 				} catch(e) {
150 					var value = (metric.type === 'text' ? '' : 0);
151 				}
152 			  switch (metric.type) {
153 			    case 'text':
154 			      $td.text(value.replace('amp;', ''));
155 			      break;
156 			    case 'value':
157 			    case 'change':
158 			    case 'pct':
159 						var decimals = 1;
160 						var suffix = '';
161 			      value = Number(value);
162 						if (metric.places !== 1) {
163 							value = value / metric.places;
164 							switch (metric.places) {
165 								case 1000:
166 									suffix = 'K';
167 									break;
168 								case 1000000:
169 									suffix = 'M';
170 									break;
171 								case 1000000000:
172 									suffix = 'B';
173 									break;
174 								case 1000000000000:
175 									suffix = 'T';
176 									break;
177 								default:
178 									suffix = '?';
179 									break;
180 							}
181 						}
182 						else {
183 							decimals = (metric.type == 'pct' ? (quote.decimals > 1 ? quote.decimals - 1 : 1) : quote.decimals);
184 						}
185 			
186 			      if (isNaN(value) || value === 0) {
187 			        $td.text('â€•');
188 			      } else {
189 							var formattedValue = addCommas(value.toFixed(decimals)) + suffix;
190 							
191 							if (metric.type !== 'value') {
192 								if (value > 0) {
193 									formattedValue = '+' + formattedValue;
194 									$td.addClass('positive');
195 								} else {
196 									$td.addClass('negative');
197 								}
198 							}
199 		          $td.text(formattedValue);
200 			      }
201 			      break;
202 			    default:
203 			      break;
204 			  }
205 				return $td;
206 			},

216 				var isPopup = (inject === undefined || inject !== true);
217 				$.each(quotesJSON, function(i, portfolio){
218 					if (isPopup || portfolio.visible) {

223 						var $tbl = $('<table>').appendTo($table).data('portfolio', portfolio.name);
224 						var $theadTr = $('<tr>').appendTo($('<thead>').appendTo($tbl));
225 						var $tfootTr = $('<tr>').appendTo($('<tfoot>').appendTo($tbl));
226 						$.each(metricsJSON, function(j, metric){
227 							if (isPopup || metric.inject) {
228 								$theadTr.append($('<td>').addClass(metric.type)
229 									.text(Display.translate(metric.name)));
230 								if (!(metric.name in sums[portfolio.name])) {
231 									sums[portfolio.name][metric.name] = new Sum(metric);
232 								}
233 							}
234 						});
235 						var $tbody = $('<tbody>').appendTo($tbl);
236 						$.each(portfolio.quotes, function(j, quote){
237 							var $tr = $('<tr>').appendTo($tbody).click(function(event){
238 								if (settingsJSON.behaviors[isPopup ? 'popup' : 'inject'] ===
239 									  'goToPage' ? !event.shiftKey : event.shiftKey) {
240 									window.open(sourcesJSON[quote.source].url + quote.secid);
241 								}
242 								else {
243 									Display.chart.draw(quote);
244 								}
245 							});
246 							$.each(metricsJSON, function(k, metric){
247 								if (isPopup || metric.inject) {
248 									$tr.append(Display.table.cell(quote, metric).data('metric', k));
249 									sums[portfolio.name][metric.name].add(quote);
250 								}
251 							});
252 						});
253 						
254 						var showFooter = false;
255 						$.each(sums[portfolio.name], function(k, sum){
256 							if (sum.nonzero()) {
257 								showFooter = true;
258 							}
259 							$tfootTr.append(Display.table.cell(sum.pseudoQuote(), sum.pseudoMetric()).data('metric', sum.name));
260 						});
261 						if (!showFooter) {
262 							$tfootTr.css('display', 'none');
263 						}
264 					}
265 				});
266 				Display.table.align();
267 				return this;
268 			},

298 							var metric = metricsJSON[metricId];

300 								.data('metric', metricId);
301 							sums[portfolio.name][metric.name].add(quote)
302 							
303 							// Compare changes in order to show highlight animation
304 							var change = compareValues($(td), $replacement);
305 							if (change !== ''){
306 								$replacement.addClass(change);
307 							}
308 							
309 							$(td).replaceWith($replacement).remove();
310 						});
311 					});
312 					$('tfoot td', table).each(function(j, td){
313 						var metricName = $(td).data('metric');
314 						var sum = sums[portfolio.name][metricName];
315 						var $replacement = Display.table.cell(sum.pseudoQuote(),
316 							sum.pseudoMetric()).data('metric', metricName);
317 						var change = compareValues($(td), $replacement);
318 						if (change !== ''){
319 							$replacement.addClass(change);							
320 						}
321 						$(td).replaceWith($replacement).remove();
322 					});
323 				});
324 				Display.table.align();
325 				return this;
326 			}
327 		},
